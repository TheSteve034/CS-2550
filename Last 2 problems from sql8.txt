SELECT t1.ENROLLED, t1.SECTION_ID, t2.CAPACITY
FROM (SELECT SECTION_ID, COUNT(*) as "ENROLLED"
FROM ENROLLMENT
GROUP BY SECTION_ID) t1 
JOIN (SELECT CAPACITY, SECTION_ID
FROM SECTION) t2 
ON t1.SECTION_ID = t2.SECTION_ID
WHERE t1."ENROLLED" >= t2.CAPACITY;
